// Function to calculate the seek time for FCFS algorithm
int fcfs(vector<int>& requests, int initialPos) {
    int seekTime = 0;
    int currentPos = initialPos;

    for (int i = 0; i < requests.size(); i++) {
        seekTime += abs(currentPos - requests[i]);
        currentPos = requests[i];
    }

    return seekTime;
}

// Function to calculate the seek time for Elevator (SCAN) algorithm
int elevator(vector<int>& requests, int initialPos) {
    int seekTime = 0;
    int currentPos = initialPos;

    sort(requests.begin(), requests.end());

    // Find the index where the current position is in the sorted requests
    auto it = lower_bound(requests.begin(), requests.end(), currentPos);

    // Move right until the end
    for (; it != requests.end(); it++) {
        seekTime += abs(currentPos - *it);
        currentPos = *it;
    }

    // Move left from the end
    for (auto it = requests.rbegin(); it != requests.rend(); it++) {
        seekTime += abs(currentPos - *it);
        currentPos = *it;
    }

    return seekTime;
}

int elevator(vector<int>& requests, int initialPos) {
    int seekTime = 0;
    int currentPos = initialPos;

    sort(requests.begin(), requests.end());

    // Find the index where the current position is in the sorted requests
    auto it = lower_bound(requests.begin(), requests.end(), currentPos);

    // Move right until the end
    for (; it != requests.end(); it++) {
        seekTime += abs(currentPos - *it);
        currentPos = *it;
    }

    // Move left from the end
    for (auto it = requests.rbegin(); it != requests.rend(); it++) {
        seekTime += abs(currentPos - *it);
        currentPos = *it;
    }

    return seekTime;
}

#include <stdio.h>
#include <stdlib.h>

// Constants
#define TOTAL_CYLINDERS 200
#define REQUESTS 10

// Function to sort requests in ascending order
void sortRequests(int requests[], int n) {
    int i, j, temp;
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (requests[i] > requests[j]) {
                temp = requests[i];
                requests[i] = requests[j];
                requests[j] = temp;
            }
        }
    }
}

// FCFS Disk Scheduling Algorithm
int fcfs(int requests[], int n, int head) {
    int seekTime = 0;
    for (int i = 0; i < n; i++) {
        seekTime += abs(head - requests[i]);
        head = requests[i];
    }
    return seekTime;
}

// Elevator Disk Scheduling Algorithm (SCAN)
int elevator(int requests[], int n, int head) {
    int seekTime = 0;
    sortRequests(requests, n);
    int direction = 1; // 1 for moving up, -1 for moving down

    for (int i = 0; i < n; i++) {
        if (direction == 1) {
            if (requests[i] >= head) {
                seekTime += requests[i] - head;
                head = requests[i];
            }
        } else {
            if (requests[i] <= head) {
                seekTime += head - requests[i];
                head = requests[i];
            }
        }

        // Change direction if needed
        if (i == n - 1 && direction == 1) {
            direction = -1;
            seekTime += head;
            head = 0;
            i = -1; // Restart at the beginning
        }
    }
    return seekTime;
}

int main() {
    int requests[REQUESTS] = {50, 98, 183, 37, 122, 14, 124, 65, 67, 34};
    int initialHead = 53;

    // FCFS
    int fcfsSeekTime = fcfs(requests, REQUESTS, initialHead);
    printf("FCFS Seek Time: %d\n", fcfsSeekTime);

    // Elevator
    int elevatorSeekTime = elevator(requests, REQUESTS, initialHead);
    printf("Elevator (SCAN) Seek Time: %d\n", elevatorSeekTime);

    return 0;
}
